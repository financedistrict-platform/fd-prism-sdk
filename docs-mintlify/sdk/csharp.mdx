## Planned API

### Basic Setup

```csharp
using Prism.AspNetCore;

var builder = WebApplication.CreateBuilder(args);

// Add Prism services
builder.Services.AddPrism(options =>
{
    options.ApiKey = builder.Configuration["Prism:ApiKey"];
    options.UseSandbox = true;
    options.Timeout = TimeSpan.FromSeconds(10);
});

var app = builder.Build();

// Use Prism middleware
app.UsePrism();

app.MapControllers();
app.Run();
```

### Controller with Attributes

```csharp
using Microsoft.AspNetCore.Mvc;
using Prism.AspNetCore.Attributes;

[ApiController]
[Route("api/[controller]")]
public class WeatherController : ControllerBase
{
    // Protected endpoint - requires payment
    [HttpGet]
    [RequirePayment(0.001, Description = "Weather data")]
    public ActionResult<WeatherForecast> Get()
    {
        return new WeatherForecast
        {
            Temperature = 72,
            Condition = "Sunny"
        };
    }

    // Premium endpoint - higher price
    [HttpGet("premium")]
    [RequirePayment(0.01, Description = "Premium weather data")]
    public ActionResult<PremiumWeather> GetPremium()
    {
        // Payment info available in HttpContext
        var payment = HttpContext.GetPaymentInfo();

        return new PremiumWeather
        {
            Temperature = 72,
            Condition = "Sunny",
            Humidity = 60,
            Pressure = 1013,
            PaymentTxHash = payment?.TxHash
        };
    }
}
```

### Minimal API

```csharp
using Prism.AspNetCore;

var app = builder.Build();

app.MapGet("/api/data", async (HttpContext context) =>
{
    return Results.Ok(new { data = "content" });
})
.RequirePayment(0.001, "Data access");

app.MapGet("/api/premium", async (HttpContext context) =>
{
    var payment = context.GetPaymentInfo();

    return Results.Ok(new
    {
        data = "premium content",
        payment = new
        {
            network = payment?.Network,
            amount = payment?.Amount,
            txHash = payment?.TxHash
        }
    });
})
.RequirePayment(0.01, "Premium data access");

app.Run();
```

### Configuration Options

```csharp
public class PrismOptions
{
    /// <summary>
    /// Required: Prism API key
    /// </summary>
    public string ApiKey { get; set; } = string.Empty;

    /// <summary>
    /// Use sandbox environment (default: true)
    /// </summary>
    public bool UseSandbox { get; set; } = true;

    /// <summary>
    /// Custom Gateway URL (overrides UseSandbox)
    /// </summary>
    public string? BaseUrl { get; set; }

    /// <summary>
    /// Request timeout (default: 10 seconds)
    /// </summary>
    public TimeSpan Timeout { get; set; } = TimeSpan.FromSeconds(10);

    /// <summary>
    /// Number of retry attempts (default: 3)
    /// </summary>
    public int Retries { get; set; } = 3;

    /// <summary>
    /// Enable detailed logging (default: false)
    /// </summary>
    public bool EnableLogging { get; set; } = false;
}
```

### Dependency Injection

```csharp
using Prism.AspNetCore.Services;

public class WeatherService
{
    private readonly IPrismClient _prismClient;
    private readonly ILogger<WeatherService> _logger;

    public WeatherService(
        IPrismClient prismClient,
        ILogger<WeatherService> logger)
    {
        _prismClient = prismClient;
        _logger = logger;
    }

    public async Task<WeatherData> GetWeatherAsync(string location)
    {
        // Manually verify payment if needed
        var payment = await _prismClient.VerifyPaymentAsync(nonce);

        if (payment.IsValid)
        {
            _logger.LogInformation(
                "Payment verified: {TxHash} on {Network}",
                payment.TxHash,
                payment.Network
            );

            return await FetchWeatherDataAsync(location);
        }

        throw new UnauthorizedException("Invalid payment");
    }
}
```

### Custom Error Handling

```csharp
using Prism.AspNetCore.Exceptions;

app.UseExceptionHandler(errorApp =>
{
    errorApp.Run(async context =>
    {
        var exception = context.Features
            .Get<IExceptionHandlerFeature>()?
            .Error;

        if (exception is PrismException prismEx)
        {
            context.Response.StatusCode = prismEx.StatusCode;
            context.Response.ContentType = "application/json";

            await context.Response.WriteAsJsonAsync(new
            {
                error = prismEx.Code,
                message = prismEx.Message,
                details = prismEx.Details
            });
        }
    });
});
```

### Testing Support

```csharp
using Prism.AspNetCore.Testing;
using Xunit;

public class WeatherControllerTests
{
    [Fact]
    public async Task Get_WithoutPayment_Returns402()
    {

        var controller = new WeatherController();

        // Act
        var result = await controller.Get();

        // Assert
        Assert.IsType<StatusCodeResult>(result);

    [Fact]
    public async Task Get_WithValidPayment_Returns200()
    {
        // Arrange
        var client = new TestPrismClient();
        client.SetupValidPayment(
            nonce: "0x123...",
            txHash: "0xabc..."
        );

        var controller = new WeatherController();

        // Act
        var result = await controller.Get();

        // Assert
        Assert.IsType<OkObjectResult>(result);
    }
}
```

## Expected Use Cases

### Financial APIs

```csharp
[ApiController]
[Route("api/finance")]
public class FinanceController : ControllerBase
{
    [HttpGet("stock/{symbol}")]
    [RequirePayment(0.005, Description = "Real-time stock data")]
    public async Task<StockQuote> GetStockQuote(string symbol)
    {
        return await _stockService.GetQuoteAsync(symbol);
    }

    [HttpGet("historical/{symbol}")]
    [RequirePayment(0.05, Description = "Historical stock data")]
    public async Task<HistoricalData> GetHistoricalData(
        string symbol,
        [FromQuery] DateTime from,
        [FromQuery] DateTime to)
    {
        return await _stockService.GetHistoricalAsync(symbol, from, to);
    }
}
```

### AI/ML APIs

```csharp
[ApiController]
[Route("api/ai")]
public class AIController : ControllerBase
{
    [HttpPost("inference")]
    [RequirePayment(0.01, Description = "AI model inference")]
    public async Task<InferenceResult> RunInference(
        [FromBody] InferenceRequest request)
    {
        var payment = HttpContext.GetPaymentInfo();

        _telemetry.TrackInference(
            model: request.ModelId,
            paymentNetwork: payment?.Network,
            amount: payment?.Amount
        );

        return await _aiService.RunInferenceAsync(request);
    }
}
```

### Content APIs

```csharp
[ApiController]
[Route("api/content")]
public class ContentController : ControllerBase
{
    [HttpGet("article/{id}")]
    [RequirePayment(0.001, Description = "Premium article")]
    public async Task<Article> GetArticle(int id)
    {
        return await _contentService.GetArticleAsync(id);
    }

    [HttpGet("video/{id}")]
    [RequirePayment(0.05, Description = "Premium video")]
    [Produces("video/mp4")]
    public async Task<IActionResult> GetVideo(int id)
    {
        var stream = await _contentService.GetVideoStreamAsync(id);
        return File(stream, "video/mp4");
    }
}
```
